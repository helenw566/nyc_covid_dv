imputed_first <- imputed_tbl %>%

      `Report Date` >= phase_one & `Report Date` < phase_two ~ 1,

Model

# run RD models

mod1 <- lm(
  phase ~ running_var_bin*indicator,
  data = imputed_first %>%
    mutate(
      indicator = ifelse(running_var_bin >= 0 , 1, 0)
    ) 
)

mod_first <- coeftest(
  mod1, 
  vcov = vcovCL(mod1, cluster = imputed_first$running_var_bin, type = "HC0")
) %>% 
  tidy() %>%
  mutate(model = "First Stage") 

# function for reduced form

reduced_model <- function(type, partner) {
  mod <- lm(
    n ~ running_var_bin*indicator,
    data = imputed_reduced %>%
      mutate(
        indicator = ifelse(running_var_bin >= 0 , 1, 0),
        n = as.numeric(n)
      ) %>%
      filter(`Offense Type` == sym(!!type), `Intimate Relationship Flag` == sym(!!partner))
  )
  
  mod_df <- coeftest(
      mod, 
      vcov = vcovCL(mod, cluster = imputed_reduced %>% 
                      filter(`Offense Type` == sym(!!type), `Intimate Relationship Flag` == sym(!!partner) ) %>% 
                      pull(running_var_bin), 
                    type = "HC0")
    ) %>%
      tidy() %>%
      mutate(
        offense_type = !!type,
        partner_flag = !!partner
      )
  
  return(mod_df)
}

# extract coef for each reduced form

model_df <- data.frame(matrix(ncol = 7, nrow = 0))

for (offense in unique(imputed_reduced$`Offense Type`)) {
  for (flag in unique(imputed_reduced$`Intimate Relationship Flag`)) {
    temp <- reduced_model(type =  offense, partner = flag)
    model_df <- rbind(model_df, temp)
  }
}
